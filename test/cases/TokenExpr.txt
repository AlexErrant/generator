@top { (word | number | operator)+ }

@skip { whitespace }

@tokens {
  whitespace { std.whitespace+ }

  word:word { std.asciiLetter (letter | std.digit)* }

  number:number { (std.digit+ ("." std.digit*)? | "." std.digit+) (("e" | "E") ("+" | "-")? std.digit+)? }

  operator:op { "+" "+"? | "-" }

  letter { [a-zA-Z] }
}

# Word tokens

Hello		Catch22 Foo
azAZ09

==> word,word,word,word

# Numbers and operators

50 + 200e-5 ++ .2 - 111.111e+111

==> number,op,number,op,number,op,number
