top { (kw<"let"> | kw<"if"> | foo)+ }

skip { whitespace }

tokens {
  whitespace { std.whitespace+ }
  kw<word>:$word.kw { word }
  foo:foo { bar<"foo"> }
  bar<x> { baz<x> }
  baz<z> { "!" z }
}

# Allows parameterized tokens

let if let

==> let.kw, if.kw, let.kw

# Tokens can call other tokens

!foo

==> foo
