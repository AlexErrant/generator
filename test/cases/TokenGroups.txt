precedence { div left }

top { expr }

expr { regExp | symbol | binOp }

binOp:op { expr !div "/" expr }

skip { whitespace }

tokens {
  whitespace { std.whitespace+ }

  symbol:sym { std.asciiLetter+ }

  regExp:re { "/" [^/]+ "/" }
}

# Disambiguates division from regexp

x / y / /foo/

==> op(op(sym,sym),re)
