program { (statement ";")+ }

statement {
  tag.Decl<kw<"let"> Identifier "=" Number> |
  tag.Print<kw<"print"> Number> |
  tag.Exit<kwSpec<"exit">> |
  tag.Async<kwExt<"async"> Number> |
  Identifier
}

kw<word> { replace<Identifier, word> }

kwSpec<word> { specialize<Identifier, word> }

kwExt<word> { extend<Identifier, word> }

tokens {
  skip { std.whitespace* }
  Identifier { std.asciiLetter+ }
  Number { std.digit+ }
}

---

let exit = 10;
exit;

==> Decl(Identifier, Number),Exit

---

print 10;
let print 2;

==> Print(Number),Decl(⚠),⚠,Print(Number)

---

async 1;
async;

==> Async(Number), Identifier
