program { (statement ";")+ }

statement {
  tag.Decl<kw<"let"> Identifier "=" Number> |
  tag.Print<kw<"print"> Number> |
  tag.Exit<kw<"exit">>
}

kw<word> { specialize<Identifier, word> }

tokens {
  skip { std.whitespace* }
  Identifier { std.asciiLetter+ }
  Number { std.digit+ }
}

---

let let = 10;
exit;

==> Decl(Identifier, Number),Exit

---

print 10;
let print = 2;

==> Print(Number),Decl(Identifier,Number)
