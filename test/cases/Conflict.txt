program { expr+ }

expr { ArrowExpr | ParenExpr | Identifier }

ParenExpr { "(" expr ")" }

ArrowExpr { "(" ParamName ")" "=>" expr }

Identifier { word ~arrow }

ParamName { word ~arrow }

skip { whitespace }

tokens {
  whitespace { std.whitespace+ }

  word { std.asciiLetter+ }
}

(a) => b

==> ArrowExpr(ParamName, Identifier)

(a)

==> ParenExpr(Identifier)
