top { expr+ }

expr { arrowExpr | parenExpr | identifier }

parenExpr:paren.expr { i<"("> expr i<")"> }

arrowExpr:arrow.expr { i<"("> paramName i<")"> i<"=>"> expr }

identifier:id.expr { word ~arrow }

paramName:param.name { word ~arrow }

skip { whitespace }

tokens {
  i<expr> { expr }

  whitespace { std.whitespace+ }

  word { std.asciiLetter+ }
}

# Arrow

(a) => b

==> arrow.expr(param.name, id)

# Paren expr

(a)

==> paren.expr(id)
