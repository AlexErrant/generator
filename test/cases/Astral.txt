top { (identifier | thing)+ }

skip { whitespace }

tokens {
  whitespace { " "+ }

  identifierChar { std.asciiLetter | [\u{a1}-\u{10ffff}] }

  identifier:id { identifierChar+ }

  thing:thing { "-" ("💩" | "🦆") }
}
