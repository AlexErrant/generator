@top { expr+ }

expr {
  ParenExpr { "(" Number ")" } |
  BracketExpr |
  WeirdExpr |
  DualExpr
}

BracketExpr {
  "[|" Number "|]"
}

WeirdExpr {
  "((" Number "()"
}

DualExpr {
  ("{" | "{{") Number ("}" | "}}")
}

@tokens { Number { std.digit+ } }

@detectDelim
