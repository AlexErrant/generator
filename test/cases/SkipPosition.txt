@top { statement+ }

statement:stat {
  variable ";" |
  variable variable ";"
}

variable:var { identifier }

@skip { space | comment }

@tokens {
  space { std.whitespace+ }
  comment:comment { "'" [^']* "'" }
  identifier { "x" }
}

# Puts skipped content in the right tree position

x'c';'c' x 'c' x 'c'; 'c'

==> stat(var, comment), comment, stat(var, comment, var, comment), comment
